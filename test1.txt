ssh-keygen -t rsa –C “email" - створити SSH ключ
git config --global user.name “ім'я/нік“ - задати ім'я користувача гіт
git config --global user.email “email" - емейл користувача гіт(показується при коммітах)
cd /c/...... - зайти в директорію

ls - що в папці
git add - додати файл з локальної машини на обробку в комміт
git commit - підтвердити комміт
git push - загрузити на гітхаб зміни
git pull - оновити з гітхаба на локальну машину
git status - статус бренча(гілки)
git log - лог змін
& - відкрити в новому вікні(ставиться в кінці команди)
git gui - відкрити "гуі" гіт баш, для удобства 
gitk - пошук коммітів, історія і тд, всяка подібна фігня
git show - інфа про комміт, хто, як чого хто батько коміта(який коміт був до того і тд)
	-s --pretty =raw ~commit name~ - також інфа про комміт 
git ls-tree (tree name) - інфа про дерево
					---відміна змін---
ДО КОМІТІВ:
	git checkout -- NAME - відмінити зміну файла на вид який був до ваших змін
		..checkout . - відмінити всі зміни всіх файлів на вид...
	git clean - видалити новостворені файли
		ФЛАГИ(можна декілька разом): 
		-x - ігнорувати правила gitignore
		-d - удалити директорії які невідомі гіту(не були ще в комітах і пушнуті)
		-f - один з обов'язкових флагів, без нього нічо не буде
